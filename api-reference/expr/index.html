    Returns:

    Type Description
    Self

    A new expression.

    Examples:

    >>> import polars as pl
>>> import pandas as pd
>>> import pyarrow as pa
>>> import narwhals as nw
>>> from narwhals.typing import IntoFrameT
>>>
>>> data = {"a": [1, -2], "b": [-3, 4]}

    diff()
